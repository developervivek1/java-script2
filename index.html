                function generateFibonacci(n) {
                    if (n <= 0) {
                        return "Number of terms must be greater than 0.";
                    }
                    const fibonacci = [0, 1]; // Start with the first two terms
                    for (let i = 2; i < n; i++) {
                        fibonacci.push(fibonacci[i - 1] + fibonacci[i - 2]);

                //12 5. Implement a function that takes a arrays and sort them without using any built-in sorting functions. 
                function customSort(array) {
                    let n = array.length;
                    for (let i = 0; i < n - 1; i++) {
                        for (let j = 0; j < n - i - 1; j++) {
                            if (array[j] > array[j + 1]) {
                                // Swap the elements
                                let temp = array[j];
                                array[j] = array[j + 1];
                                array[j + 1] = temp;
                            }
                        }
                    }
                    return array;
                }
            
                // Example usage
                const numbersSort = [64, 34, 25, 12, 22, 11, 90];
                console.log("Sorted array:", customSort(numbersSort));
                
            
                
                //13 5. Implement a function that takes two sorted arrays and merges them into a single sorted array without using any built-in sorting functions. 
                function mergeSortedArrays(arr1, arr2) {
                    let mergedArray = [];
                    let i = 0, j = 0;
            
                    // Compare elements from both arrays and add the smaller one to the merged array
                    while (i < arr1.length && j < arr2.length) {
                        if (arr1[i] < arr2[j]) {
                            mergedArray.push(arr1[i]);
                            i++;
                        } else {
                            mergedArray.push(arr2[j]);
                            j++;
                        }
                    }
            
                    // Add any remaining elements from arr1
                    while (i < arr1.length) {
                        mergedArray.push(arr1[i]);
                        i++;
                    }
            
                    // Add any remaining elements from arr2
                    while (j < arr2.length) {
                        mergedArray.push(arr2[j]);
                        j++;
                    }
            
                    return mergedArray;
                }
            
                // Example usage
                const sortedArray1 = [1, 3, 5, 7];
                const sortedArray2 = [2, 4, 6, 8];
                const result = mergeSortedArrays(sortedArray1, sortedArray2);
                console.log("Merged Sorted Array:", result);

                                let j = Math.floor(Math.random() * (i + 1)); // Random index from 0 to i

                            }

                            let right = arr.length - 1;

                                // Swap elements
                                [arr[left], arr[right]] = [arr[right], arr[left]];
                                left++;

  • What is Hoisting?

    Hoisting is a JavaScript mechanism where variable and function declarations are moved (or "hoisted") to the top of their scope (global or function scope) during the compilation phase, before the code is executed. However, only the declarations are hoisted, not the initializations.

    • ⚠️ Important: Variables declared with **var** are hoisted and initialized with undefined.
    • Variables declared with **let** and **const** are hoisted but not initialized — they are in the Temporal Dead Zone (TDZ).
    • Function declarations are hoisted with their entire body.
    • Function expressions are treated like variables (only the variable name is hoisted, not the function body).

                                    🔄 Function Borrowing in JavaScript
Function Borrowing is a JavaScript concept where an object uses a method from another object as if it were its own. This is possible using call(), apply(), or bind() to explicitly set the this context.

✅ Use Case
When two or more objects share similar properties, but only one of them has a useful method, you can borrow that method instead of redefining it.

🧠 Example: Using call() for Function Borrowing
js
Copy
Edit
const person1 = {
  name: 'Alice',
  greet: function(age) {
    console.log(`Hello, I'm ${this.name} and I'm ${age} years old.`);
  }
};

const person2 = {
  name: 'Bob'
};

// Borrow greet from person1
person1.greet.call(person2, 30);  
// Output: Hello, I'm Bob and I'm 30 years old.
✅ Example: Using apply() for Function Borrowing
js
Copy
Edit
person1.greet.apply(person2, [25]);  
// Output: Hello, I'm Bob and I'm 25 years old.
✅ Example: Using bind() for Function Borrowing
js
Copy
Edit
const greetBob = person1.greet.bind(person2, 40);
greetBob();  
// Output: Hello, I'm Bob and I'm 40 years old.

A Higher-Order Function (HOF) is a function that either:

Takes one or more functions as arguments, or

Returns a function as its result — or both.

In short, functions that operate on other functions.

✅ Common Higher-Order Functions in JavaScript:

forEach()

Custom functions that return or accept functions

Here, processUserInput is a higher-order function because it takes greet as an argument.

  • Event Loop 
                                    
The Event Loop is a core part of JavaScript’s concurrency model that allows non-blocking, asynchronous execution — even though JavaScript is single-threaded.

It enables JavaScript to perform tasks like handling timers, DOM events, and asynchronous operations (like fetch) without stopping the main thread.   


//Core Concepts Involved

    Call Stack – Where functions are executed
    Web APIs – Browser-provided APIs for async tasks (e.g., setTimeout, AJAX, DOM events)
    Callback Queue (Task Queue) – Stores callback functions waiting to be executed
    Microtask Queue – For promises, queueMicrotask(), MutationObserver, etc.
    Event Loop – The mechanism that monitors the call stack and pushes tasks from the queue when the stack is empty.


//How the Event Loop Works (Step-by-Step)
    
JS executes code in the Call Stack.
When an async operation (like setTimeout) occurs:
It's handled by Web APIs.
The callback is queued in the Callback Queue (or Microtask Queue for promises).
Once the Call Stack is empty, the Event Loop:
First checks the Microtask Queue and runs all its tasks.
Then moves one task from the Callback Queue into the stack.
Repeats the process continuously.

  • Shallow Copy and Deep Copy (structuredClone) 
                                
// Shallow Copy
=> A shallow copy creates a new object, but copies references for nested objects/arrays.

✅ Characteristics:
=> Top-level properties are copied.
=> Nested objects or arrays are shared (not cloned).
=> Changes to nested structures affect both original and copy.


// Example:

const original = {
  name: "Alice",
  address: {
    city: "Delhi"
  }
};
//
const shallowCopy = { ...original };
shallowCopy.address.city = "Mumbai";
//
console.log(original.address.city); // 🔥 Output: "Mumbai"
Both original and shallowCopy share the same address object.
                            


                            
// Deep Copy
=> A Deep Copy is a process in which all properties of an object, including nested objects and arrays, are recursively copied, resulting in a completely independent clone of the original.
=> Modifying the deep-copied object does not affect the original object, even for deeply nested structures.

//✅ Characteristics:
=> Entire structure is cloned, not just the top level.
=> No shared references — changes are independent.
=> Copies all levels of an object or array.
=> Creates new memory references for every nested element.
=> Ensures no shared references between the original and the copy.

// Example:

const original = {
  name: "Alice",
  address: {
    city: "Paris",
    zip: 12345
  }
};

// Deep copy using structuredClone (modern)
const deepCopy = structuredClone(original);

deepCopy.address.city = "Tokyo";

console.log(original.address.city); // "Paris" (unchanged)
console.log(deepCopy.address.city); // "Tokyo"

  • Currying in JavaScript 
                                    
=> Currying in JavaScript is a functional programming technique where a function with multiple arguments is transformed into a sequence of functions, each taking a single argument.
// OR
=>Currying in JavaScript is a technique of transforming a function with multiple arguments into a sequence of functions, each taking a single argument.

// 🔧 Example of Normal Function

function add(a, b) {
  return a + b;
}
add(2, 3); // 5


//🔁 Curried Version

function curriedAdd(a) {
  return function(b) {
    return a + b;
  };
}
curriedAdd(2)(3); // 5



// ✅ Benefits of Currying

=> Reusability: You can create specialized functions by partially applying arguments.
=> Function composition: Works well with other functional techniques.
=> Cleaner code when used wisely in functional patterns.

  • Rest parameter and Spread operator 
                                    
// 1. Rest Parameter (...)
Used in function parameters to collect multiple arguments into an array.

✅ Syntax:

function myFunc(...args) {
  console.log(args);
}
myFunc(1, 2, 3); // Output: [1, 2, 3]


// 📌 Example – Sum of numbers:
function sum(...numbers) {
  return numbers.reduce((acc, val) => acc + val, 0);
}
console.log(sum(1, 2, 3, 4)); // Output: 10


// 🔹 2. Spread Operator (...)
=> Used to spread elements of an array or object into individual elements.

✅ Syntax:

const arr = [1, 2, 3];
console.log(...arr); // Output: 1 2 3

// 📌 Example – Merging arrays:
const a = [1, 2];
const b = [3, 4];
const merged = [...a, ...b];
console.log(merged); // Output: [1, 2, 3, 4]

const obj1 = { name: "John" };
const obj2 = { ...obj1, age: 30 };
console.log(obj2); // Output: { name: "John", age: 30 }
  • Promises, setTimeout and Callbacks order of execution

    Understanding the execution order of Promises, setTimeout, and callbacks in JavaScript requires knowing how the JavaScript Event Loop works—particularly the call stack, microtask queue, and macrotask queue.

                                
// console.log("Start");

setTimeout(() => {
  console.log("setTimeout");
}, 0);

Promise.resolve().then(() => {
  console.log("Promise 1");
}).then(() => {
  console.log("Promise 2");
});

console.log("End");

//output

Start
End
Promise 1
Promise 2
setTimeout


// 🔎 Why This Order?
=> console.log("Start") – Synchronous → Executed first.
=> setTimeout(..., 0) – Scheduled as a macrotask.
=> Promise.then(...) – Scheduled as microtasks (added to microtask queue).
=> console.log("End") – Synchronous → Executed next.
=> Microtasks (Promise 1, then Promise 2) → Executed after sync code.
=> Macrotask (setTimeout) → Executed after all microtasks.
