                let count =0;
                function updateCount(){
                    const counter = document.querySelector('.counter');
                    counter.innerHTML = count;
                }
                function increase(){
                    count = count+1;
                    updateCount();
                }
                function decrease(){
                    if(count>0){
                        count = count-1
                    }
                    updateCount();
                }
                function reset(){
                    count = 0;
                    updateCount();
                }
