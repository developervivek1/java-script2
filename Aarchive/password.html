const password = document.querySelector('.password');
        const show = document.querySelector('.show');

        show.addEventListener('click',function(){
            
            if(password.type === 'password'){
                password.type = "text";
                show.textContent = 'Hide password';
            }else{
                password.type = "password";
                show.textContent = 'hide password';
            }
            
        })
